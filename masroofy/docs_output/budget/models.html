 45class BudgetCycle(models.Model):
 46    """Represents a user's budget cycle with allowance and daily limits."""
 47    user = models.ForeignKey(User, on_delete=models.CASCADE)
 48    objects = CycleManager()
 49
 50    total_allowance   = models.FloatField()
 51    start_date        = models.DateField()
 52    end_date          = models.DateField()
 53    remaining_balance = models.FloatField()
 54    safe_daily_limit  = models.FloatField()
 55    is_active         = models.BooleanField(default=True)
 56
 57    def __str__(self):
 58        return f"Cycle {self.id}"
 59
 60    def get_total_days(self):
 61        """Calculates the total number of days in the budget cycle.
 62
 63        Computes the duration of the cycle by subtracting start_date from end_date.
 64        This represents the planned length of the budgeting period.
 65
 66        Returns:
 67            The total number of days in the cycle as an integer.
 68            For example, a cycle from 2024-01-01 to 2024-01-31 returns 30.
 69
 70        Note:
 71            This uses date arithmetic which gives the difference in days.
 72            For a 31-day month cycle, this would return 30 (31-1).
 73        """
 74        return (self.end_date - self.start_date).days
 75
 76    def get_days_remaining(self):
 77        """Calculates the number of days remaining until the cycle ends.
 78
 79        Determines how many days are left in the current budget cycle from today.
 80        This is useful for showing progress and time pressure to users.
 81
 82        Returns:
 83            The number of days remaining as an integer. Returns a negative
 84            number if the cycle has already ended.
 85
 86        Note:
 87            Uses date.today() which may not account for timezone differences.
 88            Consider using timezone-aware dates for production applications.
 89        """
 90        from datetime import date
 91        return (self.end_date - date.today()).days
 92
 93    def get_remaining_balance(self):
 94        """Returns the current remaining balance of the budget cycle.
 95
 96        This is the amount still available to spend in the entire cycle.
 97        It starts as total_allowance and decreases with each transaction.
 98
 99        Returns:
100            The remaining balance as a float in EGP (Egyptian Pounds).
101
102        Note:
103            This value can go negative if overspending occurs, indicating
104            the cycle budget has been exceeded.
105        """
106        return self.remaining_balance
107
108    def get_spent_percentage(self):
109        """Calculates the percentage of the total allowance that has been spent.
110
111        Computes what portion of the budgeted amount has been used so far.
112        This is useful for progress indicators and budget tracking.
113
114        Returns:
115            The spent percentage as a float between 0.0 and 100.0.
116            Returns 0.0 if total_allowance is 0 to avoid division by zero.
117
118        Note:
119            Spent amount is calculated as total_allowance - remaining_balance.
120            Values over 100% indicate overspending.
121        """
122        spent = self.total_allowance - self.remaining_balance
123        if self.total_allowance == 0:
124            return 0
125        return (spent / self.total_allowance) * 100
126
127    def deduct_from_balance(self, amt):
128        """Deducts the specified amount from the cycle's remaining balance.
129
130        Reduces the available budget by the given amount and saves the change
131        to the database. This is typically called when recording expenses.
132
133        Args:
134            amt: The amount to deduct as a float. Should be positive.
135
136        Note:
137            This method modifies the model instance and saves it immediately.
138            For bulk operations, consider using update() queries instead.
139        """
140        self.remaining_balance -= amt
141        self.save()

182class DailyRecord(models.Model):
183    """Tracks daily budget allocation and spending for a budget cycle."""
184    user = models.ForeignKey(User, on_delete=models.CASCADE)
185    objects = DailyRecordManager()
186
187    date            = models.DateField()
188    allocated_limit = models.FloatField()
189    total_spent     = models.FloatField(default=0)
190    cycle           = models.ForeignKey(BudgetCycle, on_delete=models.CASCADE)
191
192    def __str__(self):
193        return f"Record {self.date}"
194
195    def get_unspent_amount(self):
196        """Calculates the unspent amount for the day.
197
198        Determines how much of the allocated daily limit remains available.
199        This is the difference between what was budgeted for the day and
200        what has actually been spent.
201
202        Returns:
203            The unspent amount as a float. Can be positive (surplus),
204            zero (spent exactly the limit), or negative (overspent).
205
206        Note:
207            Positive values can be rolled over to the next day.
208            Negative values indicate the daily limit was exceeded.
209        """
210        return self.allocated_limit - self.total_spent
211
212    def rollover_amount(self):
213        """Returns the amount available for rollover (same as unspent).
214
215        This is an alias for get_unspent_amount() for clarity in rollover logic.
216        Represents the surplus that can be carried forward to tomorrow.
217
218        Returns:
219            The rollover amount as a float. Only positive values are typically
220            rolled over; negative values are not carried forward.
221
222        Note:
223            The rollover logic ensures this amount doesn't go below zero,
224            so negative unspent amounts don't reduce future daily limits.
225        """
226        return self.get_unspent_amount()
227
228    def is_overspent(self):
229        """Checks if the daily spending has exceeded the allocated limit.
230
231        Determines whether the user has spent more than budgeted for this day.
232        This affects rollover calculations and may trigger warnings.
233
234        Returns:
235            True if total_spent > allocated_limit, False otherwise.
236
237        Note:
238            Overspending on one day can affect future daily limits through
239            deficit carryover in the rollover process.
240        """
241        return self.total_spent > self.allocated_limit

246@receiver(post_delete, sender=Transaction)
247def update_stats_on_delete(sender, instance, **kwargs):
248    """Updates cycle and daily record balances when a transaction is deleted.
249
250    Signal handler that automatically corrects budget balances when transactions
251    are removed. This ensures data consistency by reversing the balance deductions
252    that occurred when the transaction was originally recorded.
253
254    The handler performs two main operations:
255    1. Adds the transaction amount back to the cycle's remaining_balance
256    2. Subtracts the amount from the corresponding daily record's total_spent
257
258    Args:
259        sender: The Transaction model class (automatically passed by Django).
260        instance: The Transaction instance being deleted.
261        **kwargs: Additional keyword arguments from the signal.
262
263    Note:
264        This signal runs after the transaction is deleted from the database.
265        It includes safety checks to prevent negative total_spent values that
266        could occur from multiple deletions or data inconsistencies.
267
268        The cycle balance update ensures the overall budget tracking remains
269        accurate, while the daily record update maintains per-day spending accuracy.
270    """
271    # 1. Return amount to total budget
272    cycle = instance.cycle
273    cycle.remaining_balance += instance.amount
274    cycle.save()
275
276    # 2. Deduct amount from today spend
277    daily_record = DailyRecord.objects.filter(
278        cycle=cycle, 
279        date=instance.timestamp.date()
280    ).first()
281
282    if daily_record:
283        daily_record.total_spent -= instance.amount
284        # Prevent negative numbers from repeated deletes
285        if daily_record.total_spent < 0:
286            daily_record.total_spent = 0
287        daily_record.save()

247@receiver(post_delete, sender=Transaction)
248def update_stats_on_delete(sender, instance, **kwargs):
249    """Updates cycle and daily record balances when a transaction is deleted.
250
251    Signal handler that automatically corrects budget balances when transactions
252    are removed. This ensures data consistency by reversing the balance deductions
253    that occurred when the transaction was originally recorded.
254
255    The handler performs two main operations:
256    1. Adds the transaction amount back to the cycle's remaining_balance
257    2. Subtracts the amount from the corresponding daily record's total_spent
258
259    Args:
260        sender: The Transaction model class (automatically passed by Django).
261        instance: The Transaction instance being deleted.
262        **kwargs: Additional keyword arguments from the signal.
263
264    Note:
265        This signal runs after the transaction is deleted from the database.
266        It includes safety checks to prevent negative total_spent values that
267        could occur from multiple deletions or data inconsistencies.
268
269        The cycle balance update ensures the overall budget tracking remains
270        accurate, while the daily record update maintains per-day spending accuracy.
271    """
272    # 1. Return amount to total budget
273    cycle = instance.cycle
274    cycle.remaining_balance += instance.amount
275    cycle.save()
276
277    # 2. Deduct amount from today spend
278    daily_record = DailyRecord.objects.filter(
279        cycle=cycle, 
280        date=instance.timestamp.date()
281    ).first()
282
283    if daily_record:
284        daily_record.total_spent -= instance.amount
285        # Prevent negative numbers from repeated deletes
286        if daily_record.total_spent < 0:
287            daily_record.total_spent = 0
288        daily_record.save()
