  35def login_view(request):
  36    """Handles user authentication and login.
  37
  38    Processes login form submissions by authenticating the provided username
  39    and password. On successful authentication, logs the user in and redirects

  41    to the login page.

  43    Args:
  44        request: The HTTP request object containing POST data with 'username'
  45                and 'password' fields.
  46
  47    Returns:
  48        If POST request:

  54    Note:
  55        Uses Django's built-in authentication system. Error messages are

  57    """

  59        username = request.POST.get("username")

  77
  78    Args:
  79        request: The HTTP request object containing POST data with 'username',
  80                'password', 'confirm_password', and other user fields.
  81
  82    Returns:
  83        If POST request:
  84            - Redirect to 'dashboard' on successful registration
  85            - Redirect to 'signup' with error message on validation failure
  86        If GET request:
  87            - Rendered signup.html template
  88
  89    Note:
  90        Performs basic validation for username availability and password
  91        matching. Uses Django's User model for account creation.
  92    """

  94        username         = request.POST.get("username")
  95        email            = request.POST.get("email")
  96        password         = request.POST.get("password")

  98

 100            messages.error(request, "❌ Passwords do not match")

 167def _parse_cycle_form(post, today):
 168    """Parses and validates budget cycle form data from POST request.
 169
 170    Processes form input for creating or editing budget cycles, handling
 171    both duration-based (number of days) and end-date-based cycle specifications.
 172    Performs comprehensive validation and provides detailed error messages.
 173
 174    Args:
 175        post: The request.POST QueryDict containing form data with keys:
 176             - 'total_allowance': String representation of budget amount
 177             - 'duration_days': Optional string for number of days
 178             - 'end_date': Optional string in 'YYYY-MM-DD' format
 179        today: Date object representing the current date for validation.
 180
 181    Returns:
 182        Tuple of (total_allowance, end_date, errors, form_data):
 183        - total_allowance: Float budget amount or None if invalid
 184        - end_date: Date object for cycle end or None if invalid
 185        - errors: List of error message strings
 186        - form_data: Dict with original form values for re-populating forms
 187
 188    Validation Rules:
 189        - total_allowance: Must be positive float
 190        - duration_days: Must be positive integer >= 1, calculates end_date
 191        - end_date: Must be valid date >= today, or duration_days must be provided
 192        - Either duration_days or end_date must be specified
 193
 194    Note:
 195        Uses corrected date calculation (days - 1) so 30 days means
 196        day 1 through day 30 inclusive, not 31 days total.
 197    """
 198    errors    = []
 199    form_data = {}
 200
 201    raw_total = post.get("total_allowance", "").strip()
 202    raw_days  = post.get("duration_days",   "").strip()
 203    raw_end   = post.get("end_date",         "").strip()
 204    form_data.update(total_allowance=raw_total, duration_days=raw_days, end_date=raw_end)
 205
 206    total_allowance = None
 207    try:
 208        total_allowance = float(raw_total)
 209        if total_allowance <= 0:
 210            errors.append("Total budget must be greater than zero.")
 211    except (ValueError, TypeError):
 212        errors.append("Please enter a valid budget amount.")
 213
 214    end_date = None
 215    if raw_days:
 216        try:
 217            days = int(raw_days)
 218            if days < 1:
 219                errors.append("Duration must be at least 1 day.")
 220            else:
 221                # ✅ -1 يجعل 30 يوماً = اليوم الأول حتى اليوم الـ 30 (وليس الـ 31)
 222                end_date = today + timedelta(days=days - 1)
 223        except ValueError:
 224            errors.append("Please enter a valid number of days.")
 225    elif raw_end:
 226        try:
 227            end_date = datetime.strptime(raw_end, "%Y-%m-%d").date()
 228            if end_date < today:
 229                errors.append("End date must be in the future.")
 230        except ValueError:
 231            errors.append("Invalid end date format.")
 232    else:
 233        errors.append("Please specify a duration or an end date.")
 234
 235    return total_allowance, end_date, errors, form_data

 310def get_daily_alert(daily_record):
 311    """Generates alert information for daily spending status.
 312
 313    Analyzes the current day's spending against the allocated daily limit
 314    and returns appropriate alert information based on spending thresholds.
 315    Used to provide user feedback on daily budget status.
 316
 317    Args:
 318        daily_record: DailyRecord instance for the current day, or None.
 319
 320    Returns:
 321        Dict with alert information or None if no alert needed:
 322        {
 323            "msg": str,           # Alert message in Arabic
 324            "type": str,          # Bootstrap alert type ("danger", "warning", "info")
 325            "percentage": float   # Spending percentage (0-100)
 326        }
 327
 328    Alert Thresholds:
 329        - 100%+: "danger" - Daily limit reached/exceeded
 330        - 80-99%: "warning" - High spending warning
 331        - 60-79%: "info" - Moderate spending notice
 332        - <60%: None - No alert
 333
 334    Note:
 335        Queries actual transactions for the day to ensure accurate spending
 336        calculation, rather than relying on stored total_spent which may be stale.
 337    """
 338    if not daily_record:
 339        return None
 340    limit = daily_record.allocated_limit
 341    if limit <= 0:
 342        return None
 343
 344    spent = float(
 345        Transaction.objects.filter(
 346            cycle=daily_record.cycle,
 347            timestamp__date=daily_record.date,
 348        ).aggregate(total=Sum("amount"))["total"] or 0
 349    )
 350
 351    remaining  = limit - spent
 352    percentage = (spent / limit) * 100
 353
 354    if spent >= limit:
 355        return {"msg": f"Daily limit reached — spent {spent:.2f} of {limit:.2f} EGP.", "type": "danger",  "percentage": min(round(percentage), 100)}
 356    elif spent >= limit * 0.8:
 357        return {"msg": f"You've used {percentage:.0f}% of today's limit. {remaining:.2f} EGP left.", "type": "warning", "percentage": round(percentage)}
 358    elif spent >= limit * 0.6:
 359        return {"msg": f"Spending is a bit high today ({percentage:.0f}%). {remaining:.2f} EGP left.", "type": "info", "percentage": round(percentage)}
 360    return None

 363def get_cycle_alert(cycle):
 364    """Generates alert information for overall cycle budget status.
 365
 366    Analyzes total spending against the cycle's total allowance and returns
 367    appropriate alert information based on budget utilization thresholds.
 368
 369    Args:
 370        cycle: BudgetCycle instance, or None.
 371
 372    Returns:
 373        Dict with alert information or None if no alert needed:
 374        {
 375            "msg": str,           # Alert message in Arabic
 376            "type": str,          # Bootstrap alert type ("danger", "warning", "info")
 377            "percentage": float   # Spending percentage (0-100)
 378        }
 379
 380    Alert Thresholds:
 381        - 100%+: "danger" - Budget exhausted
 382        - 80-99%: "warning" - High utilization warning
 383        - 60-79%: "info" - Moderate utilization notice
 384        - <60%: None - No alert
 385
 386    Note:
 387        Calculates actual spending by querying all transactions in the cycle,
 388        ensuring accuracy even if balance calculations have discrepancies.
 389    """
 390    if not cycle:
 391        return None
 392    total = float(cycle.total_allowance)
 393    spent = float(Transaction.objects.filter(cycle=cycle).aggregate(total=Sum("amount"))["total"] or 0)
 394    remaining  = total - spent
 395    percentage = (spent / total * 100) if total > 0 else 0
 396
 397    if percentage >= 100:
 398        return {"msg": f"Budget exhausted — spent {spent:.2f} of {total:.2f} EGP.", "type": "danger", "percentage": 100}
 399    elif percentage >= 80:
 400        return {"msg": f"Heads up — {percentage:.0f}% used. {remaining:.2f} EGP left.", "type": "warning", "percentage": round(percentage)}
 401    elif percentage >= 60:
 402        return {"msg": f"You've used {percentage:.0f}% of your budget. {remaining:.2f} EGP left.", "type": "info", "percentage": round(percentage)}
 403    return None
